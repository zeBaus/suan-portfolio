// src/app/robots.txt
import type { MetadataRoute } from "next";

const BASE = process.env.NEXT_PUBLIC_SITE_URL ?? "https://example.com";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: "*",
        allow: "/",
        disallow: ["/api/", "/_next/", "/static/"]
      }
    ],
    sitemap: `${BASE}/sitemap.xml`,
    host: BASE
  };
}
